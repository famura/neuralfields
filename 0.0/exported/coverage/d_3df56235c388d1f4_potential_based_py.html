12class PotentialBased(nn.Module, ABC): 

13 """Base class for all potential-based recurrent neutral networks.""" 

14

19 self, 

20 input_size: int, 

21 hidden_size: int, 

22 activation_nonlin: Union[ActivationFunction, Sequence[ActivationFunction]], 

23 tau_init: Union[float, int], 

24 tau_learnable: bool, 

25 kappa_init: Union[float, int], 

26 kappa_learnable: bool, 

27 potentials_init: Optional[torch.Tensor] = None, 

28 output_size: Optional[int] = None, 

29 input_embedding: Optional[nn.Module] = None, 

30 output_embedding: Optional[nn.Module] = None, 

31 ): 

32 """ 

33 Args: 

34 input_size: Number of input dimensions. 

35 hidden_size: Number of neurons with potential per hidden layer. For all use cases conceived at this point, 

36 we only use one recurrent layer. However, there is the possibility to extend the networks to multiple 

37 potential-based layers. 

38 activation_nonlin: Nonlinearity used to compute the activations from the potential levels. 

39 tau_init: Initial value for the shared time constant of the potentials. 

40 tau_learnable: Whether the time constant is a learnable parameter or fixed. 

41 kappa_init: Initial value for the cubic decay, pass 0 to disable the cubic decay. 

42 kappa_learnable: Whether the cubic decay is a learnable parameter or fixed. 

43 potentials_init: Initial for the potentials, i.e., the network's hidden state. 

44 output_size: Number of output dimensions. By default, the number of outputs is equal to the number of 

45 hidden neurons. 

46 input_embedding: Optional (custom) [Module][torch.nn.Module] to extract features from the inputs. 

47 This module must transform the inputs such that the dimensionality matches the number of 

48 neurons of the neural field, i.e., `hidden_size`. By default, a [linear layer][torch.nn.Linear] 

49 without biases is used. 

50 output_embedding: Optional (custom) [Module][torch.nn.Module] to compute the outputs from the activations. 

51 This module must map the activations of shape (`hidden_size`,) to the outputs of shape (`output_size`,) 

52 By default, a [linear layer][torch.nn.Linear] without biases is used. 

53 """ 

54 # Call torch.nn.Module's constructor. 

55 super().__init__() 

56 

57 # For all use cases conceived at this point, we only use one recurrent layer. However, this variable still 

58 # exists in case somebody in the future wants to try multiple potential-based layers. It will require more 

59 # changes than increasing this number. 

60 self.num_recurrent_layers = 1 

61 

62 self.input_size = input_size 

63 self._hidden_size = hidden_size // self.num_recurrent_layers # hidden size per layer 

64 self.output_size = self._hidden_size if output_size is None else output_size 

65 self._stimuli_external = torch.zeros(self.hidden_size) 

66 self._stimuli_internal = torch.zeros(self.hidden_size) 

67 

68 # Create the common layers. 

69 self.input_embedding = input_embedding or nn.Linear(self.input_size, self._hidden_size, bias=False) 

70 self.output_embedding = output_embedding or nn.Linear(self._hidden_size, self.output_size, bias=False) 

71 

72 # Initialize the values of the potentials. 

73 if potentials_init is not None: 

74 self._potentials_init = potentials_init.detach().clone() 

75 else: 

76 if activation_nonlin is torch.sigmoid: 

77 self._potentials_init = -7 * torch.ones(1, self.hidden_size) 

78 else: 

79 self._potentials_init = torch.zeros(1, self.hidden_size) 

80 

81 # Initialize the potentials' resting level, i.e., the asymptotic level without stimuli. 

82 self.resting_level = nn.Parameter(torch.randn(self.hidden_size), requires_grad=True) 

83 

84 # Initialize the potential dynamics' time constant. 

85 self.tau_learnable = tau_learnable 

86 self._log_tau_init = torch.log(torch.as_tensor(tau_init, dtype=torch.get_default_dtype()).reshape(-1)) 

87 if self.tau_learnable: 

88 self._log_tau = nn.Parameter(self._log_tau_init, requires_grad=True) 

89 else: 

90 self._log_tau = self._log_tau_init 

91 

92 # Initialize the potential dynamics' cubic decay. 

93 self.kappa_learnable = kappa_learnable 

94 self._log_kappa_init = torch.log(torch.as_tensor(kappa_init, dtype=torch.get_default_dtype()).reshape(-1)) 

95 if self.kappa_learnable: 

96 self._log_kappa = nn.Parameter(self._log_kappa_init, requires_grad=True) 

97 else: 

98 self._log_kappa = self._log_kappa_init 

99 

100 def extra_repr(self) -> str: 

101 return f"tau_learnable={self.tau_learnable}, kappa_learnable={self.kappa_learnable}"

167 def init_hidden( 

168 self, batch_size: Optional[int] = None, potentials_init: Optional[torch.Tensor] = None 

169 ) -> Union[torch.Tensor, torch.nn.Parameter]: 

170 """Provide initial values for the hidden parameters. This usually is a zero tensor. 

171 

172 Args: 

173 batch_size: Number of batches, i.e., states to track in parallel. 

174 potentials_init: Initial values for the potentials to override the networks default values with. 

175 

176 Returns: 

177 Tensor of shape `(hidden_size,)` if `hidden` was not batched, else of shape `(batch_size, hidden_size)`. 

178 """ 

179 if potentials_init is None: 

180 if batch_size is None: 

181 return self._potentials_init.view(-1) 

182 else: 

183 return self._potentials_init.repeat(batch_size, 1) 

184 else: 

185 return potentials_init.to(device=self.device)

202 @abstractmethod 

203 def forward_one_step( 

204 self, inputs: torch.Tensor, hidden: Optional[torch.Tensor] = None 

205 ) -> (torch.Tensor, torch.Tensor): 

206 """Compute the external and internal stimuli, advance the potential dynamics for one time step, and return 

207 the model's output. 

208 

209 Args: 

210 inputs: Inputs of the current time step, of shape `(input_size,)`, or `(batch_size, input_size)`. 

211 hidden: Hidden state which are for the model in this package the potentials, of shape `(hidden_size,)`, or 

212 `(batch_size, input_size)`. Pass `None` to leave the initialization to the network which uses 

213 [init_hidden][neuralfields.PotentialBased.init_hidden] is called. 

214 

215 Returns: 

216 The outputs, i.e., the (linearly combined) activations, and the most recent potential values, both of shape 

217 `(batch_size, input_size)`. 

218 """

220 def forward(self, inputs: torch.Tensor, hidden: Optional[torch.Tensor] = None) -> (torch.Tensor, torch.Tensor): 

221 """Compute the external and internal stimuli, advance the potential dynamics for one time step, and return 

222 the model's output for several time steps in a row. 

223 

224 This method essentially calls [forward_one_step][neuralfields.PotentialBased.forward_one_step] several times 

225 in a row. 

226 

227 Args: 

228 inputs: Inputs of shape `(batch_size, num_steps, dim_input)` to evaluate the network on. 

229 hidden: Initial values of the hidden states, i.e., the potentials. By default, the network 

230 

231 Returns: 

232 The outputs, i.e., the (linearly combined) activations, and all intermediate potential values, both of 

233 shape `(batch_size, num_steps, dim_output)`. 

234 """ 

235 # Bring the sequence of inputs into the shape (batch_size, num_steps, dim_input). 

236 batch_size = PotentialBased._infer_batch_size(inputs) 

237 inputs = inputs.view(batch_size, -1, self.input_size) # moved to the desired device by forward_one_step() later 

238 

239 # If given use the hidden tensor, i.e., the potentials of the last step, else initialize them. 

240 hidden = self.init_hidden(batch_size, hidden) # moved to the desired device by forward_one_step() later 

241 

242 # Iterate over the time dimension. Do this in parallel for all batched which are still along the 1st dimension. 

243 inputs = inputs.permute(1, 0, 2) # move time to first dimension for easy iterating 

244 outputs_all = [] 

245 hidden_all = [] 

246 for input in inputs: 

247 outputs, hidden_next = self.forward_one_step(input, hidden) 

248 hidden = hidden_next.clone() 

249 outputs_all.append(outputs) 

250 hidden_all.append(hidden_next) 

251 

252 # Return the outputs and hidden states, both stacked along the time dimension. 

253 return torch.stack(outputs_all, dim=1), torch.stack(hidden_all, dim=1)
